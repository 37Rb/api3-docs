# config.json

Below is a template for an Airnode's configuration file (config.json). There are some fields that contain <FILL_*>. This means that the value added is independent from other fields. On the other hand, if two fields contain the same expression (e.g., <FILL_OPERATION_PARAMETER_1_NAME>), you must use the same value in them, because they are referencing each other. There are comments in the template JSON below that denote relationships between certain field names or values. Consider the following examples.

  • ois[0].title is related to triggers.request[0].oisTitle when they share the same field name.
  • ois[0].title could also show up in environment.securitySchemes if the ois record does in fact use a security scheme.

See config.json under Technology > Deployment Files as a reference while building a config.json file.

{
  "ois": [
    {
      "oisFormat": "1.0.0",
      "version": "<FILL_*>",
      "title": "<FILL_OIS_TITLE>",      // environment.securitySchemes[n].oisTitle
      "apiSpecifications": {            // AND triggers.request[n].oisTitle
        "info": {
          "title": "<FILL_*>"
        },
        "servers": [
          {
            "url": "<FILL_*>"
          }
        ],
        "paths": {
          "<FILL_PATH>": {
            "<FILL_METHOD>": {
              "parameters": [ 
                {
                  "in": "<FILL_OPERATION_PARAMETER_1_IN>",    // endpoints[n].fixedOperationParameters
                  "name": "<FILL_OPERATION_PARAMETER_1_NAME>" // endpoints[n].fixedOperationParameters
                },
                {
                  "in": "<FILL_OPERATION_PARAMETER_2_IN>",    // endpoints[n].parameters
                  "name": "<FILL_OPERATION_PARAMETER_2_NAME>" // endpoints[n].parameters
                }
              ]
            }
          }
        },
        "components": {
          "securitySchemes": {
            "<FILL_SECURITY_SCHEME_NAME>": {    // apiSpecifications.security[n]
              "in": "<FILL_*>",  
              "type": "<FILL_*>",
              "name": "<FILL_*>"
            }
          }
        },
        "security": [
          {
            "<FILL_SECURITY_SCHEME_NAME>": []   // environment.securitySchemes
          }                                     // AND apiSpecifications.components.securitySchemes
        ]
      },
      "endpoints": [
        {
          "name": "<FILL_ENDPOINT_NAME>",       // triggers.request[n].endpointName
          "operation": {
            "method": "<FILL_METHOD>",
            "path": "<FILL_PATH>"
          },
          "fixedOperationParameters": [
            {
              "operationParameter": {                        
                "in": "<FILL_OPERATION_PARAMETER_1_IN>",    
                "name": "<FILL_OPERATION_PARAMETER_1_NAME>"
              },
              "value": "<FILL_*>"
            }
          ],
          "reservedParameters": [
            {
              "name": "<FILL_*>",
              "fixed": "<FILL_*>"
            },
            {
              "name": "<FILL_*>",
              "fixed": "<FILL_*>"
            },
            {
              "name": "<FILL_*>",
              "default": "<FILL_*>"
            },
            {
              "name": "<FILL_*>"
            }
          ],
          "parameters": [
            {
              "name": "<FILL_*>",
              "default": "<FILL_*>",
              "operationParameter": {
                "in": "<FILL_OPERATION_PARAMETER_2_IN>",
                "name": "<FILL_OPERATION_PARAMETER_2_NAME>"
              }
            }
          ]
        }
      ]
    }
  ],
  "triggers": {
    "request": [
      {
        "endpointId": "<FILL_*>",              
        "oisTitle": "<FILL_OIS_TITLE>",        // ois[n].title
        "endpointName": "<FILL_ENDPOINT_NAME>" // ois[n].apiSpecifications.endpoints[n].name
      }
    ]
  },
  "nodeSettings": {
    "cloudProvider": "aws",
    "logFormat": "plain",
    "logLevel": "INFO",
    "nodeVersion": "0.1.0",
    "region": "<FILL_*>",
    "stage": "<FILL_*>"
  },
  "chains": [
    {
      "airnodeAdmin": "<FILL_*>",
      "authorizers": [
        "<FILL_*>"
      ],
      "contracts": {
        "AirnodeRrp": "<FILL_*>"
      },
      "id": "<FILL_CHAIN_ID_1>",               // environment.chainProviders[n].chainId
      "providerNames": [
        "<FILL_PROVIDER_NAME_1>"               // environment.chainProviders[n].name
      ],
      "type": "<FILL_CHAIN_TYPE_1>"            // environment.chainProviders[n].chainType
    }
  ],
  "environment": {
    "chainProviders": [
      {
        "chainType": "<FILL_CHAIN_TYPE_1>",    // chains[n].type
        "chainId": "<FILL_CHAIN_ID_1>",        // chains[n].id
        "name": "<FILL_PROVIDER_NAME_1>",      // chains[n].providerNames[n]
        "envName": "<FILL_*>"
      }
    ],
    "securitySchemes": [
      {
        "oisTitle": "<FILL_OIS_TITLE>",        // ois[n].title
        "name": "<FILL_SECURITY_SCHEME_NAME>", // ois[n].apiSpecifications.security[n]
        "envName": "<FILL_*>"
      }
    ]
  }
}
